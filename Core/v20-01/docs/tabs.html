function OpenTabContent(evt, id) {
    let tab_cont = document.getElementsByClassName("m3-tab-content");
    for (i = 0; i < tab_cont.length; i++) {
        tab_cont[i].classList.remove("active");
    }

    let tabs = document.getElementsByClassName("m3-tab");
    for (i = 0; i < tabs.length; i++) {
        tabs[i].classList.remove("active");
    }
    
    document.getElementById(id).classList.add("active");
    evt.currentTarget.classList.add("active");
}

Custom Indicator

Add your own set of rules for .m3-tab.active


.m3-tab.active{
    border-bottom: unset;
    border-radius: 10px;
    background: rgba(62, 178, 199, 0.432);
}

Active by Default

Option #1: Add the class .active on both, .m3-tab and .m3-tab-content


<div class="m3-tabs">
    <button class="m3-tab active" onclick="OpenTabContent(event, 'AnyID1');">Active by Default</button>
</div>

<section class="m3-tab-content active" id="AnyID1">
    Your content goes here
</section>

Option #2: Use JavaScript to programmatically click on a tab. 


<div class="m3-tabs">
    <button class="m3-tab" id='TabID' onclick="OpenTabContent(event, 'AnyID1');">Active by Default</button>
</div>

<section class="m3-tab-content" id="AnyID1">
    Your content goes here
</section>

<script>
    /* IMPORTANT: Add this after the declaration of the function OpenTabContent() */
    document.getElementById("TabID").click();
</script>
